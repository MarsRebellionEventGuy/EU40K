					days = 90
				}
			}

			factor = 1
			modifier = {
				factor = 2
				NOT = {
					OR = {
						has_country_modifier = priesthood_infiltrated
						FROM = { ai = yes } 
					}
				}		
			}
		}
	}

	option = {
		name = admech_vassalisation.1.c
		hidden_effect = {
			IF = {
				limit = {
					NOT = { development = 50 }

			}

					AND = {

					}

			}

				limit = {
					development = 200
				}

			}

				}

			factor = 5
		}
	}
