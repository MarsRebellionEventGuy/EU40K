country_event = {
	id = admech_vassalisation.1
	title = "admech_vassalisation.1.t"
	desc = "admech_vassalisation.1.desc"
	picture = NOBLE_ESTATE_DEMANDS_eventPicture
	
	is_triggered_only = yes

	trigger = {
		has_country_flag = admech_has_been_offered_subjugation_trigger 
	}

	immediate = {
		clr_country_flag = admech_has_been_offered_subjugation_trigger 
	}
	#become a vassal
	option = {
		name = admech_vassalisation.1.a
		hidden_effect = {
			FROM = {
				country_event = { 
					id = admech_vassalisation.2 
					#days = 10						#test
					days = 90
				}
			}
		}
		ai_chance = {
			factor = 6
			modifier = {
				factor = 0.5
				NOT = {
					OR = {
						has_country_modifier = priesthood_infiltrated
						FROM = { ai = yes } 
					}
				}		
			}
		}
	}
	#remain independent
	option = {
		name = admech_vassalisation.1.b
		hidden_effect = {
			FROM = {
				country_event = { 
					id = admech_vassalisation.3
					#days = 10						#test
					days = 90
				}
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 2
				NOT = {
					OR = {
						has_country_modifier = priesthood_infiltrated
						FROM = { ai = yes } 
					}
				}		
			}
		}
	}

	#negotiate - we will make demands
	option = {
		name = admech_vassalisation.1.c
		hidden_effect = {
			IF = {
				limit = {
					NOT = { development = 50 }
				}
				set_country_flag = admech_vassalisation_s
			}
			IF = {
				limit = {
					AND = {
						NOT = { development = 200 }
						development = 50
					}
				}
				set_country_flag = admech_vassalisation_m
			}
			IF = {
				limit = {
					development = 200
				}
				set_country_flag = admech_vassalisation_l
			}
			set_country_flag = admech_vassalisation_dummy_1
			FROM = {
				country_event = {
					id = admech_vassalisation.100
				}
			}
		}
		ai_chance = {
			factor = 5
		}
	}
}

								has_province_modifier = forge_world_2_area_mod
								has_province_modifier = forge_world_3_area_mod
								has_province_modifier = forge_world_4_area_mod
							}
						}
						2={
							change_variable = {
								which = MechanicusNbForgeWorlds
								value = 1
