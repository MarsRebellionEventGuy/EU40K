namespace = ai_forge_world

#start
#target
country_event = {
	id = ai_forge_world.1
	title = "ai_forge_world.1.title"
	desc = "ai_forge_world.1.desc"
	picture = NOBLE_ESTATE_DEMANDS_eventPicture
	
	hidden = yes
	
	trigger = {
		AND = {
			primary_culture = mechanicus_culture
			OR = {
				any_owned_province = {
					has_province_modifier = fallen_forge_world_mod
				}
				any_owned_province = {
					has_province_modifier = fallen_major_forge_world_mod
				}
			}
			is_at_war = no
			ai = yes
		}
	}
	mean_time_to_happen = {
		months = 120
		
		#is a major forge world
		modifier = {
			factor = 0.9
			OR = {
				any_owned_province = {
					has_province_modifier = forge_world_1_mod
				}
				any_owned_province = {
					has_province_modifier = forge_world_2_mod
				}
				any_owned_province = {
					has_province_modifier = forge_world_3_mod
				}
				any_owned_province = {
					has_province_modifier = forge_world_4_mod
				}
			}
		}
		
		#has 2+ fallen forge worlds, to speed up restoration
		modifier = {
			factor = 0.9
			calc_true_if = {
				OR = {
					all_owned_province = {
						has_province_modifier = fallen_forge_world_mod
					}
					all_owned_province = {
						has_province_modifier = fallen_major_forge_world_mod
					}
				}
				amount = 2
			}
		}
		
		#has 4+ forge worlds, more ressources for repairs
		modifier = {
			factor = 0.9
			calc_true_if = {
				OR = {
					all_owned_province = {
						has_province_modifier = minor_forge_world_mod
					}
					all_owned_province = {
						has_province_modifier = forge_world_1_mod
					}
					all_owned_province = {
						has_province_modifier = forge_world_2_mod
					}
					all_owned_province = {
						has_province_modifier = forge_world_3_mod
					}
					all_owned_province = {
						has_province_modifier = forge_world_4_mod
					}
				}
				amount = 4
			}
		}
		
		#has loans
		modifier = {
			factor = 1.2
			num_of_loans = 1
		}
	}

	immediate = {
	}
	
	option = {
		name = ai_forge_world.1.a
		
		if = {
			limit = {
				any_owned_province = {
					has_province_modifier = fallen_major_forge_world_mod
				}
				save_event_target_as = target_restore_forge_world
			}
			event_target:target_restore_forge_world = {
				remove_province_modifier = fallen_major_forge_world_mod
				add_permanent_province_modifier = {
					name = forge_world_1_mod
					duration = -1
				}
			}
			forge_world_restoration_cost = yes
		}
		
		else_if = {
			limit = {
				any_owned_province = {
					has_province_modifier = fallen_forge_world_mod
				}
				save_event_target_as = target_restore_forge_world
			}
			event_target:target_restore_forge_world = {
				remove_province_modifier = fallen_forge_world_mod
				add_permanent_province_modifier = {
					name = minor_forge_world_mod
					duration = -1
				}
			}
			forge_world_restoration_cost = yes
		}
	}
}

#"free" dev for AI in capital
country_event = {
	id = ai_forge_world.5
	title = "ai_forge_world.5.title"
	desc = "ai_forge_world.5.desc"
	picture = NOBLE_ESTATE_DEMANDS_eventPicture
	
	hidden = yes
	
	trigger = {
		AND = {
			primary_culture = mechanicus_culture
			capital_scope = {
				OR = {
					AND = {
						has_province_modifier = forge_world_1_mod
						NOT = { development = 30 }
					}
					AND = {
						has_province_modifier = forge_world_2_mod
						NOT = { development = 40 }
					}
					AND = {
						has_province_modifier = forge_world_3_mod
						NOT = { development = 50 }
					}
				}
			}
			is_at_war = no
			ai = yes
		}
	}
	mean_time_to_happen = {
		months = 100
		
		modifier = {
			factor = 0.9
			stability = 2
		}
	}
	
	option = {
		name = ai_forge_world.5.a
		capital_scope = {
			add_base_tax = 1
		}
		add_adm_power = -30
		ai_chance = { factor = 1 }
	}
	
	option = {
		name = ai_forge_world.5.b
		capital_scope = {
			add_base_production = 1
		}
		add_dip_power = -30
		ai_chance = { factor = 1 }
	}
	
	option = {
		name = ai_forge_world.5.c
		capital_scope = {
			add_base_manpower = 1
		}
		add_mil_power = -30
		ai_chance = { factor = 1 }
	}
	
}
