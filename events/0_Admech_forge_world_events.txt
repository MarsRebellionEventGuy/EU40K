		is_forge_world_trigger = yes
		owner = {
			culture = mechanicus_culture
		}

				}

				add_prestige = 10

					limit = {
						NOT = { OR = {
							has_country_modifier = forge_world_2_subject_limit
							has_country_modifier = forge_world_3_subject_limit
							has_country_modifier = forge_world_4_subject_limit
						}		}
					}

		}

			owner = { 
				add_prestige = 20
