# country
on_leader_recruited = {
	if = {
		limit = { has_faction = astartes_faction_1 }
		add_faction_influence = {
			faction = astartes_faction_2
			influence = 10
		}
	}
}

on_alliance_created = { #ROOT - Alliance sender, FROM - Alliance accepter
	if = {
		limit = { has_faction = astartes_faction_1 }
		add_faction_influence = {
			faction = astartes_faction_2
			influence = 25
		}
	}
	if = {
		limit = { FROM = { has_faction = astartes_faction_1 } }
		FROM = {
			add_faction_influence = {
				faction = astartes_faction_2
				influence = 25
			}
		}
	}
}
