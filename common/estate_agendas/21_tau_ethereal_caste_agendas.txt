estate_ethereal_caste_restore_devotion = {
	can_select = {
		has_dlc = "Common Sense"
		NOT = { legitimacy_equivalent = 65 }
		government = theocracy
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 1.5
			NOT = { legitimacy_equivalent = 50 }
		}
		modifier = {
			factor = 1.5
			NOT = { legitimacy_equivalent = 35 }
		}
	}
	fail_if = {
		NOT = { government = theocracy }
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_ethereal_caste
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_ethereal_caste
			}
		}
		else = {
			legitimacy_equivalent = 90
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_ethereal_caste
		}
		add_estate_loyalty = {
			estate = estate_ethereal_caste
			loyalty = 10
		}
		generate_scaled_advisor_of_type_and_religion_effect = {
			type = theologian
			religion = root
			discount = yes
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_ethereal_caste
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

		}
		modifier = {
			factor = 0.5

				type = theologian
				religion = root
			}
		}

			factor = 2
			advisor = theologian
			NOT = {
				employed_advisor = {
					type = theologian
					religion = root
				}
			}
		}
	}

		hidden_effect = {
			set_country_flag = need_to_fire_adm_advisor

				}

			has_estate_agenda_auto_completion = {
				estate = estate_ethereal_caste
			}
		}
		else = {
			custom_trigger_tooltip = {
				tooltip = fire_adm_advisor_tooltip
				has_country_flag = just_fired_adm_advisor
			}
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_ethereal_caste
		}
		add_estate_loyalty = {
			estate = estate_ethereal_caste
			loyalty = 10
		}
		clr_country_flag = just_fired_adm_advisor
		clr_country_flag = need_to_fire_adm_advisor
	}
	invalid_trigger = {
		NOT = {
			employed_advisor = {
				category = ADM
			}
		}

		on_failed_agenda_effect = yes

            estate = estate_ethereal_caste
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		clr_country_flag = need_to_fire_adm_advisor
    }
}

#Root out heresy in X
estate_ethereal_caste_root_out_heresy = {
	can_select = {
		cb_on_religious_enemies = yes
		is_free_or_tributary_trigger = yes
		num_of_cities = 10
               
		any_country = {
            border_distance = {
                who = root
                distance = 200
            }
			NOT = { religion = root }
			NOT = { num_of_cities = 6 }
			is_free_or_tributary_trigger = yes
			NOT = { alliance_with = root }
			NOT = { marriage_with = root }
			NOT = { war_with = root }
			NOT = { ROOT = { truce_with = prev } }
			ROOT = {
				NOT = {
					has_opinion = {
						who = prev
						value = 100
					}
				}
			}
			NOT = { army_size = root }
		}
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 2
			has_idea_group = religious_ideas
		}
		modifier = {
			factor = 0.35
			has_idea_group = humanist_ideas
		}
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_country = {
			limit = {
                border_distance = {
                    who = root
                    distance = 200
                }
				NOT = { religion = root }
				NOT = { num_of_cities = 6 }
				is_free_or_tributary_trigger = yes
				NOT = { alliance_with = root }
				NOT = { marriage_with = root }
				NOT = { war_with = root }
				NOT = { ROOT = { truce_with = prev } }
				ROOT = {
					NOT = {
						has_opinion = {
							who = prev
							value = 100
						}
					}
				}
				NOT = { army_size = root }
			}
			save_event_target_as = agenda_country
			capital_scope = {
				save_event_target_as = agenda_country_capital
			}
		}
	}
	fail_if = {
		OR = {
			is_subject_other_than_tributary_trigger = yes
			has_country_modifier = recently_changed_religion_dummy
			event_target:agenda_country = {
				is_subject_other_than_tributary_trigger = yes
				NOT = { is_subject_of = root }
			}
			AND = {
				NOT = {
					event_target:agenda_country = {
						num_of_cities = 1
					}
				}
				NOT = {
					event_target:agenda_country_capital = {
						owned_by = root #you annexed them
					}
				}
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_country_capital
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_ethereal_caste
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_ethereal_caste
			}
		}
		else = {
			hidden_trigger = { NOT = { has_country_modifier = recently_changed_religion_dummy } }
			if = {
				limit = {
					event_target:agenda_country = {
						num_of_cities = 1
					}
				}
				event_target:agenda_country = {
					religion = root
				}
			}
			else = {
				event_target:agenda_country_capital = {
					owned_by = root #you annexed them
				}
			}
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_ethereal_caste
		}
		add_estate_loyalty = {
			estate = estate_ethereal_caste
			loyalty = 10
		}
		generate_scaled_advisor_of_type_and_religion_effect = {
			type = inquisitor
			religion = root
			discount = yes
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_ethereal_caste
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Ulema: Propagate religion in trade node (if you have CoC)
estate_ethereal_caste_propagate_religion = {
	can_select = {
		has_dlc = "Cradle of Civilization"
		religion = greater_good
		dominant_religion = root
		any_active_trade_node = {
			trade_share = {
				country = root
				share = 50
			}
			is_node_in_trade_company_region = yes
			any_trade_node_member_province = {
				is_city = yes
				NOT = { religion = greater_good }
			}
			calc_true_if = {
				amount = 10
				all_trade_node_member_province = {
					religion = root
				}
			}
		}
		NOT = { has_country_modifier = missionary_zeal }
	}
	selection_weight = {
		factor = 2.5
	}
	pre_effect = {
		random_active_trade_node = {
			limit = {
				trade_share = {
					country = root
					share = 50
				}
				is_node_in_trade_company_region = yes
				any_trade_node_member_province = {
					is_city = yes
					NOT = { religion = greater_good }
				}
				calc_true_if = {
					amount = 10
					all_trade_node_member_province = {
						religion = root
					}
				}
			}
			save_event_target_as = agenda_trade_node
		}
	}
	provinces_to_highlight = {
		any_trade_node_member_province = {
			province_id = event_target:agenda_trade_node
		}
		is_city = yes
		NOT = { religion = greater_good }
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_ethereal_caste
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_ethereal_caste
			}
		}
		else = {
			custom_trigger_tooltip = {
				tooltip = estate_ethereal_caste_propagate_religion_trigger_tooltip
				event_target:agenda_trade_node = {
					all_trade_node_member_province = {
						OR = {
							religion = root
							is_city = no
						}
					}
				}
			}
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_ethereal_caste
		}
		add_estate_loyalty = {
			estate = estate_ethereal_caste
			loyalty = 15
		}
		add_country_modifier = {
			name = missionary_zeal
			duration = 5475
		}
	}
	fail_if = {
		OR = {
			NOT = { religion_group = muslim }
			event_target:agenda_trade_node = {
				all_trade_node_member_province = {
					NOT = {
						trade_share = {
							country = root
							share = 50
						}
					}
				}
			}
		}
	}
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_ethereal_caste
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}
