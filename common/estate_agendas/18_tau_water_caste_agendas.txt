			}

		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_water_caste
				}
			}
			has_estate_agenda_auto_completion = {

			}
		}

			if = {
				limit = {
					check_variable = {

					}

		add_estate_loyalty = {
			estate = estate_water_caste

		}
