					country = root
					share = 20
				}
			}
			NOT = {
				any_country = {
					event_target:agenda_trade_node = {
						privateer_power = {
							share = 3
							country = PREV
						}
					}
				}
			}
		}
	}
	fail_if = {
		event_target:agenda_trade_node = {
			NOT = {
				trade_share = {
					country = root
					share = 10
				}
			}
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_air_caste
		}
		add_estate_loyalty = {
			estate = estate_air_caste
			loyalty = 15
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_air_caste
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
