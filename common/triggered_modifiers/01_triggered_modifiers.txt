# Triggered modifiers are here.
# these are checked for each countries once/month and then applied.
#
# Effects are fully scriptable here.

missing_mechanicus_modifier = {
	potential = {
		is_emperor_of_china = yes
		NOT={
			any_subject_country={
				is_subject_of_type = 0_mechanicus_subject_type
			}
		}
	}

	trigger = {
		NOT={
			any_subject_country={
				is_subject_of_type = 0_mechanicus_subject_type
			}
		}
	}

	global_trade_goods_size_modifier = -0.25
	imperial_mandate = -0.05
}
primarch_regent = {
	potential = {
		is_emperor_of_china = yes
	}

	trigger = {
		ruler_has_personality = primarch_personality
	}

	ai_weight = {
		factor = -1000 # Do what you can to avoid this modifier
	}

	imperial_mandate = 0.02
}
lost_control_of_tera = {
	potential = {
		is_emperor_of_china = yes
	}

	trigger = {
		OR = {
			NOT = { owns = 1 }
			NOT = { controls = 1 }
		}
	}

	ai_weight = {
		factor = -1000 # Do what you can to avoid this modifier
	}

	imperial_mandate = -0.05
}
