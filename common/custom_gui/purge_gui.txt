custom_button = {
	name = purge_button
	potential = {
		OR = {
			AND = {
				owner = {
					culture_group = tau_culture_group
				}
				has_owner_accepted_culture = no
			}
			AND = {
				owner = {
					NOT = { culture_group = tau_culture_group }
				}
				OR = {
					NOT = { culture_group = owner }
					NOT = { religion_group = owner }
				}
			}
		}
	}
	trigger = {
		owner = { 
			if = {
				limit = { has_estate_privilege = estate_inquisition_right_exterminatus }
				mil_power = 50
			}
			else_if = {
				limit = { has_reform = final_solution_reform }
				mil_power = 33
			}
			else = {
				mil_power = 100 
			}
			OR={
				TAG = FROM
				is_subject_of = FROM
			}
		}
	}
	effect = {
		owner = { 
			if = {
				limit = { has_estate_privilege = estate_inquisition_right_exterminatus }
				add_mil_power = -50
				PREV = { add_devastation = 10 }
			}
			else_if = {
				limit = {has_reform = final_solution_reform }
				add_mil_power = -33
				PREV = { add_devastation = 100 }
			}
			else = {
				add_mil_power = -100
			}
		}
		spawn_rebels = {
			size = 4
			leader = noble_zealot
			type = nationalist_rebels
		}
		change_culture = owner
		change_religion = owner
		add_province_modifier = {
			name = province_purged
			duration = 7000
		}
	}
	tooltip = purge_button_tt
}
