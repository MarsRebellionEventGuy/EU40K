
caste_privilege_counter_on_granted_effect = {
	if = {
		limit = {
			NOT = { has_country_flag = $CASTE$_privilege_count_1 }
		}
		set_country_flag = $CASTE$_privilege_count_1
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_1
		}
		clr_country_flag = $CASTE$_privilege_count_1
		set_country_flag = $CASTE$_privilege_count_2
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_2
		}
		clr_country_flag = $CASTE$_privilege_count_2
		set_country_flag = $CASTE$_privilege_count_3
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_3
		}
		clr_country_flag = $CASTE$_privilege_count_3
		set_country_flag = $CASTE$_privilege_count_4
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_4
		}
		clr_country_flag = $CASTE$_privilege_count_4
		set_country_flag = $CASTE$_privilege_count_5
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_5
		}
		clr_country_flag = $CASTE$_privilege_count_5
		set_country_flag = $CASTE$_privilege_count_6
	}
}

caste_privilege_counter_on_revoked_effect = {
	if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_1
		}
		clr_country_flag = $CASTE$_privilege_count_1
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_2
		}
		clr_country_flag = $CASTE$_privilege_count_2
		set_country_flag = $CASTE$_privilege_count_1
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_3
		}
		clr_country_flag = $CASTE$_privilege_count_3
		set_country_flag = $CASTE$_privilege_count_2
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_4
		}
		clr_country_flag = $CASTE$_privilege_count_4
		set_country_flag = $CASTE$_privilege_count_3
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_5
		}
		clr_country_flag = $CASTE$_privilege_count_5
		set_country_flag = $CASTE$_privilege_count_4
	}
	else if = {
		limit = {
			has_country_flag = $CASTE$_privilege_count_6
		}
		clr_country_flag = $CASTE$_privilege_count_6
		set_country_flag = $CASTE$_privilege_count_5
	}
}
