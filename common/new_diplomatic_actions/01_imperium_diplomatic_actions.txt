# Elevate Nominal to Imperial status
reign_in_nominal_world = {
	category = influence

	alert_index = 5
	alert_tooltip = reign_in_eyalet_alert_tooltip

	require_acceptance = yes

	is_visible = {
		is_emperor_of_china = yes
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = nominal_imperial_world_subject_type
		}
	}
	is_allowed = {
		if = {
			limit = {
				FROM = { liberty_desire = 50 }
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_REBELLIOUS_TT
				FROM = { liberty_desire = 50 }
			}
		}
		if = {
			limit = { NOT = { has_global_flag = eoc_vassalize_tributaries } } 
			custom_trigger_tooltip = {
				tooltip = IMPERIAL_REINTEGRATION_NOT_PASSED_TT
				has_global_flag = eoc_vassalize_tributaries 
			}
		}
		imperial_mandate = 5
	}
	on_accept = {
		create_subject = {
			subject_type = imperial_world_subject_type
			subject = FROM
			keep_liberty_desire = yes
		}
		add_mandate = -5
		1={
			change_variable = {
				which = ImperiumNbNominalWorlds
				value = -1
			}
			change_variable = {
				which = ImperiumNbWorlds
				value = 1
			}
		}
	}
	on_decline = {
	}

	ai_will_do = { # this is a trigger
		always = yes	#There is never a reason for the Ottomans to not want to integrate their eyalets
	}

	ai_acceptance = {
        add_entry = {
            name = DIPLO_OPPINION
			export_to_variable = {
				variable_name = ai_value
				value = opinion
				who = FROM
				with = ROOT
			}
			multiply_variable = {
				which = ai_value 
				value = 0.2
			}
        }
		add_entry = {
			name = AI_TRUST_TOWARDS_ROOT
			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = { 
				which = ai_value
				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
				which = ai_value
				value = 0.4
			}
		}
		add_entry = {
			name = AI_DIPLOMATIC_REPUTATION
			export_to_variable = {
				variable_name = ai_value
				value = trigger_value:diplomatic_reputation
			}
			#1 rep = 4
			multiply_variable = {
				which = ai_value
				value = 4
			}
		}
		add_entry = {
			name = AI_TOTAL_DEVELOPMENT
			export_to_variable = {
				variable_name = ai_value
				value = total_development
				who = FROM
			}
			#4 dev = -1
			multiply_variable = {
				which = ai_value
				value = -0.25
			}
		}
		add_entry = {
			name = HAS_BONUS_REIGN_IN_REASON
			limit = {
				has_country_flag = tur_reduced_requirements_for_core_eyalet
			}
			export_to_variable = {
				variable_name = ai_value
				value = 1
			}
			multiply_variable = {
				which = ai_value
				value = 20
			}
		}
		add_entry = {
			name = AI_LIBERTY_DESIRE
			export_to_variable = {
				variable_name = ai_value
				value = liberty_desire
				who = FROM
			}
			multiply_variable = {
				which = ai_value
				value = -1
			}
		}
	}
}

call_astartes_into_wars = {
	category = alliance

	require_acceptance = no

	is_visible = {
		is_emperor_of_china = yes
		FROM = {
			is_subject_of = ROOT
			is_subject_of_type = autonomus_astartes_subject_type
		}
	}
	is_allowed = {
		if = {
			limit = {
				FROM = { liberty_desire = 50 }
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_REBELLIOUS_TT
				FROM = { liberty_desire = 50 }
			}
		}
		if = {
			limit = {
				is_at_war = no
			}
			custom_trigger_tooltip = {
				tooltip = ROOT_IS_NOT_AT_WAR_TT
				is_at_war = yes
			}
		}
		if = {
			limit = {
				FROM = {
					NOT = {
						owes_favors = {
							who = ROOT
							value = 10
						}
					}
				}
				is_in_war = {
					defender_leader = ROOT
				}
			}
			custom_trigger_tooltip = {
				tooltip = FROM_MUST_OWE_10_FAVORS_DEFENSIVE_WAR_TT
				FROM = {
					owes_favors = {
						who = ROOT
						value = 10
					}
				}
			}
		}
		if = {
			limit = {
				FROM = {
					NOT = {
						owes_favors = {
							who = ROOT
							value = 20
						}
					}
				}
				NOT = {
					is_in_war = {
						defender_leader = ROOT
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = FROM_MUST_OWE_20_FAVORS_TT
				FROM = {
					owes_favors = {
						who = ROOT
						value = 20
					}
				}
			}
		}
		if = {
			limit = {
				FROM = { is_at_war = yes }
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_AT_WAR_TT
				FROM = { is_at_war = no }
			}
		}
		if = {
			limit = {
				all_war_enemy_countries={
					truce_with = FROM
				}
			}
			custom_trigger_tooltip = {
				tooltip = FROM_NO_TRUCE_WITH_FOES_TT
				any_war_enemy_country={
					NOT={truce_with = FROM}
				}
			}
		}
	}
	on_accept = {
		FROM = {
			ROOT = {
				add_favors = {
					who = PREV
					amount = -10
				}
			}
			join_all_defensive_wars_of = ROOT
		}
		if = {
			limit = {
				FROM = {
					owes_favors = {
						who = ROOT
						value = 10
					}
				}
			}
			FROM = {
				ROOT = {
					add_favors = {
						who = PREV
						amount = -10
					}
				}
				join_all_offensive_wars_of = ROOT
			}
		}
	}
	on_decline = {
	}

	ai_will_do = { # this is a trigger
		always = yes
	}
}

# nominal/autonomus Astartes joining imperial
invite_into_imperium = {
	category = influence
	require_acceptance = yes

	alert_index = 60
	alert_tooltip = invite_into_cultural_sphere_of_influence_alert_tooltip

	is_visible = {
		is_emperor_of_china = yes
		is_subject = no
		FROM = { 
			is_subject = no 
			culture_group = imperial_culture_group
			religion_group = emperor_group
			NOT = {religion = mechanicus }
		}
	}
	is_allowed = {
		if = {
			limit = {
				NOT = {
					FROM = {
						capital_scope = {
							trade_range = ROOT
						}
					}
				}
			}
			custom_trigger_tooltip = {
				tooltip = REQUIRES_CAPITAL_IN_TRADE_RANGE_TT
				FROM = {
					capital_scope = {
						trade_range = ROOT
					}
				}
			}
		}
		if = {
			limit = {
				FROM = { is_at_war = yes }
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_AT_WAR_TT
				FROM = { is_at_war = no }
			}
		}
		if = {
			limit = {
				is_at_war = yes
			}
			custom_trigger_tooltip = {
				tooltip = ROOT_IS_AT_WAR_TT
				is_at_war = no
			}
		}
		if = {
			limit = { 
				FROM = { 
					NOT = { religion = imperial_cult } 
					NOT = { religion = imperial_truth } 
					NOT = { religion = promethean_cult } 
					NOT = { religion = death_cult } 
				} 
			}
			custom_trigger_tooltip = {
				tooltip = FROM_IS_NOT_ROOT_RELIGION_TT
				FROM = { religion_group = root }
			}
		}
		if = {
			limit = { FROM = { is_great_power = yes } }
			custom_trigger_tooltip = {
				tooltip = FROM_IS_GREAT_POWER_TT
				FROM = { is_great_power = no }
			}
		}
		if = {
			limit = { FROM = { total_development = root } }
			custom_trigger_tooltip = {
				tooltip = FROM_MORE_DEV_THAN_ROOT_TT
				FROM = { NOT = { total_development = root } }
			}
		}
	}
	on_accept = {
		if={#Astartes Chapter
			limit={
				FROM={
					has_reform = astartes_reform
				}
			}
			create_subject = {
				subject_type = autonomus_astartes_subject_type
				subject = FROM
			}
			1={
				change_variable = {
					which = ImperiumNbIndAstartes
					value = 1
				}
			}
		}
		else_if={#Deathwatch Hold
			limit={
				FROM={
					has_reform = death_watch_reform
				}
			}
			event_target:inquisition_tag={
				create_subject = {
					subject_type = deathwatch_subject_type
					subject = FROM
				}
			}
			3942={
				change_variable = {
					which = InquisitionNbDeathwatch
					value = 1
				}
			}
		}
		else={#Independant World
			create_subject = {
				subject_type = nominal_imperial_world_subject_type
				subject = FROM
			}
			1={
				change_variable = {
					which = ImperiumNbNominalWorlds
					value = 1
				}
				change_variable = {
					which = ImperiumNbTotalWorlds
					value = 1
				}
			}
		}
		add_mandate = 2
	}
	on_decline = {
	}

	ai_will_do = { # this is a trigger
		OR = {
			ai_attitude = {
				who = FROM
				attitude = attitude_friendly
			}
			ai_attitude = {
				who = FROM
				attitude = attitude_protective
			}
			FROM = {
				any_neighbor_country = {
					NOT = { tag = ROOT }
					FROM = {
						ai_attitude = {
							who = PREV
							attitude = attitude_threatened
						}
					}
				}
			}
			FROM = {
				ai_attitude = {
					who = ROOT
					attitude = attitude_threatened
				}
			}
		}
		NOT = {
			ai_attitude = {
				who = FROM
				attitude = attitude_domineering
			}
		}
		NOT = {
			ai_attitude = {
				who = FROM
				attitude = attitude_hostile
			}
		}
		NOT = {
			ai_attitude = {
				who = FROM
				attitude = attitude_outraged
			}
		}
		has_opinion = {
			who = FROM
			value = 50
		}

			}

        }
		add_entry = {

			export_to_variable = {
				variable_name = ai_value
				value = trust
				who = FROM
				with = ROOT
			}			
			subtract_variable = {

				value = 50
			}
			#1 trust = +4 ai points
			multiply_variable = {
