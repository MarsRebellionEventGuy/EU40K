				estate_influence = {

				}
			}
			NOT = { horde_unity = 70 } 
		}

				}

		}

		}

			factor = 0.5

			}

			factor = 1

			}
		}

			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_synapses
					loyalty = 20
				}
			}
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
